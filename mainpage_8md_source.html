2 # What is CUSP?
3 <blockquote>
4 Cusp is a library for sparse linear algebra and graph computations based on Thrust. Cusp provides a flexible, high-level interface for manipulating sparse matrices and solving sparse linear systems.

22 # Example
23 ~~~{.cpp}
24 #include <cusp/hyb_matrix.h>
25 #include <cusp/io/matrix_market.h>
26 #include <cusp/krylov/cg.h>
27 
28 int main(void)
29 {
30  // create an empty sparse matrix structure (HYB format)
31  cusp::hyb_matrix<int, float, cusp::device_memory> A;
32 
33  // load a matrix stored in MatrixMarket format
34  cusp::io::read_matrix_market_file(A, "5pt_10x10.mtx");
35 
36  // allocate storage for solution (x) and right hand side (b)
37  cusp::array1d<float, cusp::device_memory> x(A.num_rows, 0);
38  cusp::array1d<float, cusp::device_memory> b(A.num_rows, 1);
39 
40  // solve the linear system A * x = b with the Conjugate Gradient method
41  cusp::krylov::cg(A, x, b);
42 
43  return 0;
44 }
